  -v my_volume:/data \
  -v $(pwd):/backup \
  ubuntu tar czf /backup/volume.tar.gz /data
--remote backup: copia su un altro server, e.g. rsync -av /srv user@backup-server:/backups/
--backup automatici (cron): si settano auto e.g.ogni giorno alle 2am.
